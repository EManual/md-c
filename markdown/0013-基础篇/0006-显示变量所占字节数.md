    #[Tags]/* char型的字节数为1 */

    #[Tags]/* short型的字节数为2 */
    #printf(#[Fields]"short:%d bytes\n"#,sizeof(short));
    #[Tags]/* long型的字节数为4 */
    #printf(#[Fields]"long:%d bytes\n"#,sizeof(#[Keywords]long#));

    #printf(#[Fields]"float:%d bytes\n"#,sizeof(#[Keywords]float#));
    #[Tags]/* double型的字节数为8 */
